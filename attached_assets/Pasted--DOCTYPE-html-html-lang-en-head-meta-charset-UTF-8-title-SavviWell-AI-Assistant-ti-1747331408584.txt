    // Demo interaction handling
    document.addEventListener('DOMContentLoaded', function() {
      // Generate buttons
      document.getElementById('generateBtn1').addEventListener('click', function() {
        // Simulate loading
        this.textContent = "Generating...";
        this.disabled = true;
        
        // Show response after a short delay to simulate AI generation
        setTimeout(() => {
          document.getElementById('response1').classList.remove('hidden');
          this.classList.add('hidden');
        }, 1500);
      });
      
      document.getElementById('generateBtn2').addEventListener('click', function() {
        this.textContent = "Generating...";
        this.disabled = true;
        setTimeout(() => {
          document.getElementById('response2').classList.remove('hidden');
          this.classList.add('hidden');
        }, 1500);
      });
      
      document.getElementById('generateBtn3').addEventListener('click', function() {
        this.textContent = "Generating...";
        this.disabled = true;
        setTimeout(() => {
          document.getElementById('response3').classList.remove('hidden');
          this.classList.add('hidden');
        }, 1500);
      });
      
      // Waitlist buttons
      const joinWaitlistButtons = document.querySelectorAll('[id^="joinWaitlist"]');
      joinWaitlistButtons.forEach(button => {
        button.addEventListener('click', function() {
          document.getElementById('demo-container').classList.add('hidden');
          document.getElementById('waitlist-container').classList.remove('hidden');
        });
      });
      
      document.getElementById('backToDemo').addEventListener('click', function() {
        document.getElementById('waitlist-container').classList.add('hidden');
        document.getElementById('demo-container').classList.remove('hidden');
      });
      
      // Waitlist form submission
      document.getElementById('waitlist-form').addEventListener('submit', function(e) {
        e.preventDefault();
        // Simulate form submission
        document.getElementById('confirmation-message').classList.remove('hidden');
        document.getElementById('email').value = '';
      });
    });
